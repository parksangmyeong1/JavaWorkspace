// 시작페이지 설정 $('#startPage').on('click', function() { var agent = navigator.userAgent.toLowerCase(); if ((navigator.appName == 'Netscape' && agent.indexOf('trident') != -1) || (agent.indexOf("msie") != -1)) { // ie일 경우 if (agent.indexOf("msie") != -1) { document.body.style.behavior = 'url(#default#homepage)'; document.body.setHomePage('http://www.naver.com'); } else { alert("internet explorer(10버전 이하) 에서만 지원하는 기능입니다."); } } else { // ie가 아닐 경우 alert("지원되지 않는 브라우저입니다."); } });
